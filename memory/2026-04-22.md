### Technical Design

#### Color Resolution Algorithm
```javascript
function getSeatColor(seat) {
  const colors = seatColors.filter(c => c.app === 'boxoffice' && c.page === 'stadium-maintenance')
  
  // 1. Exact match: type + code + status
  const exactMatch = colors.find(c => 
    c.type && c.code && c.status && 
    ((c.type === 'restriction' && c.code === seat.restriction_code) ||
     (c.type === 'category' && c.code === seat.category_code)) &&
    c.status === seat.status
  )
  if (exactMatch) return exactMatch.color
  
  // 2. Type + code match (any status)
  const typeCodeMatch = colors.find(c => 
    c.type && c.code && !c.status &&
    ((c.type === 'restriction' && c.code === seat.restriction_code) ||
     (c.type === 'category' && c.code === seat.category_code))
  )
  if (typeCodeMatch) return typeCodeMatch.color
  
  // 3. Status match (default type with status)
  const statusMatch = colors.find(c => 
    (!c.type || c.type === 'default') && !c.code && c.status && c.status === seat.status
  )
  if (statusMatch) return statusMatch.color
  
  // 4. Default available/blocked
  const defaultMatch = colors.find(c => 
    (!c.type || c.type === 'default') && !c.code && !c.status
  )
  if (defaultMatch) return defaultMatch.color
  
  // Fallback
  return seat.status === 'blocked' ? 'var(--color-danger-light)' : 'var(--color-secondary)'
}
```
